start_time = time.time()

def insertion_sorting(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        n = i - 1
        while n >= 0 and key < arr[n]:
            arr[n + 1] = arr[n]
            n -= 1
        arr[n + 1] = key

import time

start_time = time.time()

def bubble_sort(arr):
    n = len(arr)

    for i in range(n):
        for x in range(0, n - i - 1):
            if arr[x] > arr[x + 1]:
                arr[x], arr[x + 1] = arr[x + 1], arr[x]

my_list = [3, 7, 1, 9, 5, 2, 8]
bubble_sort(my_list)
print("Відсортований масив: ", my_list)
print("--- %s seconds ---" % (time.time() - start_time))

import timeit
import random
import matplotlib.pyplot as plt

sizes = [5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000, 10000, 20000, 50000, 100000]
num_trials = 10  # Кількість прогонів для кожного розміру масиву

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for x in range(0, n - i - 1):
            if arr[x] > arr[x + 1]:
                arr[x], arr[x + 1] = arr[x + 1], arr[x]
    return arr

def insertion_sorting(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        n = i - 1
        while n >= 0 and key < arr[n]:
            arr[n + 1] = arr[n]
            n -= 1
        arr[n + 1] = key
    return arr

insertion_times = []
bubble_times = []

for size in sizes:
    array = random.sample(range(1, size + 1), size)
    
    print(f"\nМасив на ({size}) елементів")

    insertion_sorted_array = insertion_sorting(array.copy())
    insertion_time = timeit.timeit(stmt=lambda: insertion_sorting(array.copy()), number=num_trials)
    insertion_avg_time = insertion_time / num_trials
    insertion_times.append(insertion_avg_time)
    print(f"Сортування вставкою: {insertion_avg_time:.6f} секунди середнього часу")


    bubble_sorted_array = bubble_sort(array.copy())
    bubble_time = timeit.timeit(stmt=lambda: bubble_sort(array.copy()), number=num_trials)
    bubble_avg_time = bubble_time / num_trials
    bubble_times.append(bubble_avg_time)
    print(f"Сортування бульбашкою: {bubble_avg_time:.6f} секунди середнього часу")


plt.figure(figsize=(10, 6))
plt.plot(sizes, insertion_times, label='Сортування вставкою')
plt.plot(sizes, bubble_times, label='Сортування бульбашкою')
plt.xlabel('Розмір масиву')
plt.ylabel('Середній час виконання (секунди)')
plt.title('Час виконання сортувань')
plt.legend()
plt.show()

Відповідь на Контрольні запитання
1) O - нотація вказує на межу часової або просторової складності алгоритму.
    Якщо функція f(n)=O(g(n)), то f(n) обмежується зверху константою, помноженою на g(n).
    Ω - нотація вказує на нижню межу часової або просторової складності алгоритму.
    Якщо функція f(n)=Ω(g(n)), то f(n) обмежується знизу константою, помноженою на g(n).
2) Сортування методом вставки має оптимальну часову складність вже у випадку, коли масив відсортований.
    Такий випадок відомий як "найсприятливіша умова" для алгоритму сортування методом вставки.
    Оптимальна часова складність у цьому випадку дорівнює O(n), де "n" - кількість елементів у масиві.
3) Ефективний алгоритм - основна ідея алгоритму полягає в тому щоб досягти максимально можливого результату при мінімальній витраті ресурсів.
     Основні критерії ефективності алгоритму включають:
      - Часова складність
      - Просторова складність
      - Робота в найгіршому випадку
      - Спрощення та оптимізація коду
      - Маштабованість
4) Поняття головного параметру задачі вказує на розмір вхідних даних, які обробляє алгоритм. Він визначає розмірність задачі та впливає на час чи ресурси, які потрібні для вирішення цієї задачі.
     Функція часової складності має вигляд: $F(N)=N^3+7N^2-14N$.
     Щоб виразити асимптотичну складність використовуємо "O", домінантом є $N^3$,
     звідси можемо надати відповідь, що асимптотична складність буде записана, як $O(N^3)$
